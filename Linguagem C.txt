Linguagem C é Case Sensitive , ou seja , maiúsculas e minúsculas se diferenciam
Cada linha de código fecha com ponto e vírgulo e chaves

// As linhas abaixo "chamam" as bibliotecas

// Entrada e saída do sistema
#include <stdio.h>

// Funções básicas do sistema
#include <stdlib.h>

// Comentários são ignorados pelo aplicativo

/* comentários em bloco */ 
// comentário


// int Números inteiros de –32768 a 32767
// main ou principal é o ponto de partida
Dentro do main teremos parâmetros
// argc é número de linhas do código
// char é um tipo de variável que aceita qualquer coisa ( Caracter , letra , número , etc )
// argv é o conteúdo das linhas do código
// return 0 é uma função de retorno ao main
// 0 indica que o aplicativo retornou sem erros no código

int main (int argc, char *argv[]
{
    printf("Hello World");
    printf("Hello World \n");
    printf(\n "Hello World \n");
    system(''pause'');
    return 0;
}

// printf = Escreve na tela o texto
// \n = pula uma linha
// system = Pede pra você apertar algo
